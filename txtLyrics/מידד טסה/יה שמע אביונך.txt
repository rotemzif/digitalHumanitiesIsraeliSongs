יָהּ שְׁמַע אֶבְיוֹנֶיךָ 
הַמְחַלִּים פָּנֶיךָ 
אָבִינוּ לְבָנֶיךָ 
אַל תַּעְלֵם אָזְנֶךָ 
יָהּ עַם מִמַּעֲמַקִּים 
יִקְרְאוּ מֵרוֹב מְצוּקִים 
אַל נָא תְּשִׁיבֵם רֵקִים 
הַיּוֹם מִלְּפָנֶיךָ 
הַוּוֹתָם וַעֲוֹנָם 
מְחֵה וְרֻבֵּי זְדוֹנָם 
וְאִם לֹא תַעֲשֶׂה לְמַעֲנָם 
עֲשֵׂה צוּרִי לְמַעֲנֶךָ 
וּמְחֵה הַיּוֹם חוֹבָם 
וּרְצֵה כְּמוֹ שַׁי נִיבָם 
וּלְךָ תָּכִין לִבָּם 
וְגַם תַּקְשִׁיב אָזְנֶךָ 
דִּמְעַת פְּנֵיהֶם תִשְׁעֶה 
וְתֶאֱסוֹף עֵדֶר תּוֹעֶה 
וְתָקִים לְךָ רוֹעֶה 
וּפְקוֹד בְּטוֹב צֹאנֶךָ 
הוֹלְכֵי בְּדֶרֶךְ נְכֹחָה 
תְּבַשְּׂרֵם הַיּוֹם סְלִיחָה 
וּבִתְפִלַּת הַשָחַר
הַמְצִיאֵם חִנֶּךָ
